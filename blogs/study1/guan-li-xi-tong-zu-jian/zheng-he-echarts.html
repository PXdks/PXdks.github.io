    整合echarts

    # 整合图表echarts

    页面整合echarts

    npm install echarts --save //下载
    1

    使用echarts (opens new window)可视化图表库，使页面的数据更为丰富多彩清晰明了。

    前端引入echarts

    import * as echarts from 'echarts'; //引入

var chartDom = document.getElementById('main'); //获取一个id为main的div
var myChart = echarts.init(chartDom); //初始化
var option;

option = {
  xAxis: {
    type: 'category',
    data: ['Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat', 'Sun'] //x轴数据， 数据可自定义，一般由后端接口改造传入
  },
  yAxis: {
    type: 'value'
  },
  series: [
    {
      data: [150, 230, 224, 218, 135, 147, 260], //y轴数据
      type: 'line'//类型 line线型，bar柱型，pie饼型
    }
  ]
};

 myChart.setOption(option); //设置

    后端改造数据接口

    例如传入各个季度的注册人数

    package com.example.test.controller;

import cn.hutool.core.collection.CollUtil;
import cn.hutool.core.date.DateUtil;
import cn.hutool.core.date.Quarter;
import com.example.test.entity.Time;
import com.example.test.service.TimeService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;
import java.sql.Timestamp;
import java.util.Map;

@RestController
@RequestMapping("/echarts")
public class EchartsController {
    @Autowired
    private TimeService timeService;
    @GetMapping("/members")
    public List members()
    {
        List<Time> list=timeService.list();
        int q1=0;
        int q2=0;
        int q3=0;
        int q4=0;
        for (Time time : list)
        {
            Timestamp createtime = time.getCreatetime();
            Quarter quarter = DateUtil.quarterEnum(createtime);
            switch (quarter) {
                case Q1:q1+=1;break;
                case Q2:q2+=1;break;
                case Q3:q3+=1;break;
                case Q4:q4+=1;break;
                default:break;
            }
            }
        return CollUtil.newArrayList(q1,q2,q3,q4); //返回一个数组类型
    }
}
