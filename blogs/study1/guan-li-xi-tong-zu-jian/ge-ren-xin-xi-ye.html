    # 个人信息页

    在用户成功登陆时，在用户信息页展示该用户的所有信息。

    具体实现

    <template>
  <div>
    <div class="header">
      Hi! {{username}},欢迎进入 <span>{{message}}开始新的一天,现在是{{newTime}}</span>
    </div>
<el-divider></el-divider>
    <div class="container"><span style="padding: 10px 10px 20px 0">个人信息</span>
      <div >
        <div class="cards">
          <!-- A card with given width -->
          <div class="cards__item">
          姓名
            <div class="message">  {{ username }}</div>
          </div>

          <!-- Repeat other cards -->
          <div class="cards__item">
           电话
            <div class="message">this is tel</div>
          </div>
          <div class="cards__item">
            邮箱
            <div class="message">this is email</div>
          </div>
          <div class="cards__item">
            住址
            <div class="message">this is address</div>
          </div>
          <div class="cards__item">
            所属部门
            <div class="message">this is department</div>
          </div>
          <div class="cards__item">
            头像
            <div class="message">
              <el-upload
                  action="http://localhost:9090/file/upload"
                  :show-file-list="false"
                  :on-success="handleAvatarSuccess"
                 >
                <img :src="avatarUrl" v-if="avatarUrl" class="logo" title="点击修改图片">
                <i v-else class="el-icon-plus avatar-uploader-icon"></i>

              </el-upload>
            </div>
          </div>
        </div>

      </div>
    </div>
  </div>


</template>

<script>
import request from "@/utils/request";

const days = ['天', '一', '二', '三', '四', '五', '六']; // 星期数组
var icnow = new Date();      // 初始化时间
var interval;                // 定义全局定时器，用于清除定时器

export default {
  name: "PersonInfo",
  data()
  {
    return{
      avatarUrl:'',
      year: icnow.getFullYear(),
      month: icnow.getMonth() + 1,
      date: icnow.getDate(),
      day: days[icnow.getDay() - 1],
      time: icnow.toTimeString().substring(0, 8),
      username:"",
      message:'11',
      hours:new Date().getHours(),
    }
  },
  created() {
    this.avatarUrl=JSON.parse(localStorage.getItem("user")).avatar;
    this.username=JSON.parse(localStorage.getItem("user")).username;
    interval = setInterval(() =>{
      let icnow = new Date();
      this.year = icnow.getFullYear();
      this.month = icnow.getMonth() + 1;
      this.date = icnow.getDate();
      this.day = days[icnow.getDay()];
      this.time = icnow.toTimeString().substring(0, 8);
    }, 1000)
    request.get("/admin/select", {
      params:{
        username:this.username
      }
    }).then(res=>{
      console.log("这是用户头像"+res.avatar)
      this.avatarUrl=res.avatar;
      localStorage.setItem("avatar",this.avatarUrl)
    })
    if(this.hours<12)
    {
      this.message="上午好 !"
    }
    if(this.hours>=12&&this.hours<=18)
    {
      this.message="下午好!"
    }
    else {
      this.message="晚上好!"
    }
  },
  computed: {
    // 当前时间
    newTime: function () {
      return this.year + '年' + this.month + '月' + this.date + '日 星期' + this.day + ' ' + this.time;
    }
  },
  beforeDestroy () {
    clearInterval(interval);
  },
  methods:{
    handleAvatarSuccess(res)
    {
      this.avatarUrl=res;
      this.username=JSON.parse(localStorage.getItem("user")).username;
      console.log(res);
      request.get("/admin/update",{
        params:{
          avatar:this.avatarUrl,
          username:this.username,
        }
      }).then(res=>{
        //触发refreshAva方法进行同步更新头像
        this.$emit("refreshAva")
        console.log(res);
      })
    }
  }

}
</script>
