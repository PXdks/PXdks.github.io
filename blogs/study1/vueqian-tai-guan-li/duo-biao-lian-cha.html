    # 多表联查

    在学生教师选课系统进行联合查询

    有ROLE_STUDENT(学生)和ROLE_TEACHER(老师)两种身份

    学生可以选择已有的课程

    并且可以在个人信息中查看已选课程

    老师可以编辑课程，发布(是否启用)课程

    并且在个人信息里查看

    这样就完成了课程表，权限表，用户表的多表联查

            <result column="teacherCourseName" property="name"></result>

            <result column="teacherTimes" property="times"></result>
        </collection>
    //学生选课集合
    <collection property="stuCourse" javaType="java.util.ArrayList" ofType="com.example.test.entity.Course">
        <result column="stuCourseName" property="name"></result>
        <result column="stuScore" property="score"></result>
        <result column="stuTimes" property="times"></result>
    </collection>

</resultMap>
    //多表查询sql语句
    <select id="findPage"  resultMap="pageAdmin">
--         select admin.*,course.* from admin
-- left join course
-- on admin.id=course.teacher_id
            select admin.*,sc.name as stuCourseName,tc.name as teacherCourseName,tc.score as teacherScore,
            sc.score  as stuScore, tc.times as teacherTimes, sc.times as stuTimes from admin
            left join student_course
            on admin.id=student_course.student_id
            left join course sc
            on student_course.course_id=sc.id
            left join course tc
            on admin.id=tc.teacher_id

    </select>

</mapper>
