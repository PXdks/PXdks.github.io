    # 增删查改

    在启动类同级目录下创建 controller 包，新建 TimeController(负责操作)，再创建 mapper 包下新建 TimeMapper(接口)

    @RestController //请求controller注解
@RequestMapping("/time")//接口路径
public class TimeController {
    @Autowired //注入字段
    private TimeMapper timeMapper;
    @Autowired
    private TimeService timeService;
    @GetMapping //获取
    public List<Time> index()
    {
    List all= timeMapper.findAll();
        return all;
    }
    @PostMapping //更新或新增的请求
    //@RequestBody注解将json转化为对象格式
    public  Integer save(@RequestBody Time time)
    {
        //更新或新增
        return timeService.save(time);
    }
    @DeleteMapping("/{id}") //删除的请求
    public Integer delete(@PathVariable Integer id)
    {
        return timeMapper.deleteById(id);
    }
}

    # 更新与新增

    创建 service 业务包，新建 TimeService 类

    @Service //service业务逻辑注解
public class TimeService {
    @Autowired
    TimeMapper timeMapper;
    public int save(Time time)
    {
        if (time.getId()==null)//id 为空为新增
        {
          return   timeMapper.insert(time);
        }
        else {//否则为更新
         return    timeMapper.update(time);
        }
    }
}

    # 分页查询

    给定参数pageNum(第几页)pageSize(多少条数据)

    sql语句为

    SELECT *FROM time LIMIT 0,4
---第一页
---（2-1）*4=4
---第二页
SELECT *FROM time LIMIT 4,4
---结论：(pageNum-1)*pageSize

     Integer total=  timeMapper.selectTotal();
